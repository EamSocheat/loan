CREATE TABLE tbl_currency (
		cur_id	int NOT NULL AUTO_INCREMENT, 
		cur_nm	varchar(30) not null,
		cur_nm_kh	varchar(30) not null,
		cur_syn	varchar(10) not null,
		cur_syn_kh	varchar(10) not null,
		regDt	datetime,
		upDt	datetime,
		useYn	varchar(1) not null,
		regUsr	int not null, 
		upUsr	int not null,
		com_id	int not null,
		PRIMARY KEY(cur_id) 
);


CREATE TABLE tbl_contract (
		con_id	int NOT NULL AUTO_INCREMENT, 
		con_start_dt datetime not null,
		con_end_dt datetime not null,
		con_principle double not null,
		con_interest double not null,
		con_period	int not null,
		con_emi	double not null,
		con_total_principle double not null,
		con_total_interest double not null,
		con_status varchar(10) not null,
		con_type int not null,
		regDt	datetime,
		upDt	datetime,
		useYn	varchar(1) not null,
		regUsr	int not null, 
		upUsr	int not null,
		com_id	int not null,
		PRIMARY KEY(con_id) 
);

CREATE TABLE tbl_installment (
		ins_id	int NOT NULL AUTO_INCREMENT, 
		ins_num	int not null,
		ins_dt datetime not null,
		ins_principle double not null,
		ins_interest double not null,
		ins_status	varchar(10) not null,
		ins_pay_dt datetime,
		con_total_interest double not null,
		con_status double not null,
		regDt	datetime,
		upDt	datetime,
		useYn	varchar(1) not null,
		regUsr	int not null, 
		upUsr	int not null,
		com_id	int not null,
		PRIMARY KEY(ins_id) 
);
